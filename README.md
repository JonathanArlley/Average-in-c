#include <stdio.h>
#include <stdlib.h>

int main()
{
    float note1 = 0;
    float note2 = 0 ;
    float note3 = 0 ;
    float average = 0;

    printf("Enter your first note: ");
    scanf("%f", &note1);

    printf("Enter your second note: ");
    scanf("%f", &note2);

    printf("Enter your third note: ");
    scanf("%f", &note3);

    average = (note1 + note2 + note3) / 3;

    printf("Your average is: %f", average);

    return 0;
}
